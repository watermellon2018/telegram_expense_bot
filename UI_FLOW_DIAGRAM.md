invite_create_{project_id}_{role}       # Create with role
role_change_{proj}_{member}_{new_role}  # Change member role
member_kick_{proj}_{member_id}          # Initiate kick
kick_confirm_{proj}_{member_id}         # Confirm kick
leave_confirm_{project_id}              # Confirm leave
member_info_{proj}_{member_id}          # Member details

# For: role_change_1_67890_viewer
